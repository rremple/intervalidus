case class DiscreteInterval2D[T1, T2](horizontal: DiscreteInterval1D[T1], vertical: DiscreteInterval1D[T2])(implicit evidence$1: DiscreteValue[T1], evidence$2: DiscreteValue[T2]) extends DiscreteIntervalLike[DiscreteDomain2D[T1, T2], DiscreteInterval2D[T1, T2]]
